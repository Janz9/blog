WebDriver API

WebDriver 使用浏览器供应商提供的浏览器自动化 API 来控制浏览器并运行测试, 就像真实用户操作浏览器一样, 由于 WebDriver 不需要使用应用程序代码来编译其 API, 因此它不是侵入式的

WebDriver API 是用于操作 WebDriver 的命令集

浏览器操作

/* 示例: 浏览器操作 */
@Test
public void test() {

    // 打开页面
    driver.get("http://news.anhuai.com/");
    // 跳转页面
    driver.navigate().to("http://news.anhuai.com/user/login");

    // 后退操作
    driver.navigate().back();
    // 前进操作
    driver.navigate().forward();
    // 刷新操作
    driver.navigate().refresh();

    // 窗口最大化
    driver.manage().window().maximize();
    // 窗口高宽
    driver.manage().window().setSize(new Dimension(1000, 500));
    // 窗口全屏
    driver.manage().window().fullscreen();

    // 获取标题
    String title = driver.getTitle();
    // 获取当前URL
    String url = driver.getCurrentUrl();
    // 获取源码
    String source = driver.getPageSource();

    // 关闭当前窗口
    driver.close();
}

元素操作

/* 示例: 元素操作 */
@Test
public void test() {

    driver.get("http://news.anhuai.com/user/login");

    // 获取元素
    WebElement username = driver.findElement(By.id("username"));
    WebElement password = driver.findElement(By.id("password"));
    WebElement submit = driver.findElement(By.className("ant-btn"));

    // 获取元素属性
    String placeholder = username.getAttribute("placeholder");
    if (StringUtils.isNotBlank(placeholder)) {
        // 清空文本
        username.clear();
    }

    // 输入文本
    username.sendKeys("135xxxxxxxx");
    password.sendKeys("xxxxxxxx");

    // 判断元素是否可点击
    if (submit.isDisplayed()) {
        // 点击元素
        submit.click();
    }

    driver.close();
}

鼠标键盘

/* 示例: 鼠标键盘 */
@Test
public void test() {

    driver.get("http://news.anhuai.com/user/login");
    WebElement username = driver.findElement(By.id("username"));
    WebElement password = driver.findElement(By.id("password"));

    // 使用Actions类实现鼠标操作
    Actions action = new Actions(driver);
    action.click(username) // 单击
        .doubleClick(username) // 双击
        .contextClick() // 右击
        .moveToElement(username) // 移动到元素
        .pause(3000) // 悬浮3秒
        .clickAndHold() // 单击并按住不放
        .release() // 释放按键
        .dragAndDrop(username, password) // 拖曳
        .perform() // 执行Actions链中存储的行为
        ;

    // 使用Keys类实现键盘操作
    username.sendKeys("135xxxxxxxxX" + Keys.BACK_SPACE); // 退格键
    username.sendKeys(Keys.CONTROL, "a"); // 全选CTRL+A
    username.sendKeys(Keys.CONTROL, "c"); // 复制CTRL+C
    password.sendKeys(Keys.CONTROL, "v"); // 粘贴CTRL+V

    driver.close();
}

超时处理

/* 示例: 超时处理 */
@Test
public void test() {

    driver.get("http://news.anhuai.com/user/login");

    // 强制等待3秒
    Thread.sleep(3000);

    /// 隐式等待
    // 设置页面加载的超时时间
    driver.manage().timeouts().pageLoadTimeout(3, TimeUnit.SECONDS);
    // 设置获取元素的超时时间
    driver.manage().timeouts().implicitlyWait(3, TimeUnit.SECONDS);
    // 设置异步脚本的超时时间
    driver.manage().timeouts().setScriptTimeout(3, TimeUnit.SECONDS);

    /// 显式等待
    // 设置超时时间为3秒, 每500毫秒调用一次判断
    WebDriverWait wait = new WebDriverWait(driver, 3L, 500L);
    // 设置判断条件, 此处判断元素是否存在
    wait.until(driver2 -> driver2.findElement(By.id("username")));
    // 获取元素
    WebElement username = driver.findElement(By.id("username"));

    driver.close();
}

窗口处理

/* 示例: 多窗口处理 */
@Test
public void test() {

    driver.get("http://news.anhuai.com/");
    driver.get("http://news.anhuai.com/user/login");

    // 获取当前窗口的句柄
    String handle = driver.getWindowHandle();

    // 获取所有窗口的句柄
    Set<String> handles = driver.getWindowHandles();
    handles.forEach(h -> {
        // 切换到指定窗口
        driver.switchTo().window(h);
        System.out.println(driver.getTitle());
        driver.close();
    });
}
/* 示例: iframe窗口处理 */
@Test
public void test() {

    driver.get("http://i.firefoxchina.cn/");
    WebElement words = driver.findElement(By.className("engine-link-words"));

    // 切换到第一个iframe标签
    driver.switchTo().frame(0);
    // 切换到指定iframe元素
    driver.switchTo().frame(words);

    // 退出iframe返回当前页面
    driver.switchTo().defaultContent();
    
    driver.close();
}
/* 示例: 弹窗处理 */
@Test
public void test() {

    /// 示例基于原生弹窗
    // Alert / Confirm / Prompt

    // 切换到弹窗
    Alert alert = driver.switchTo().alert();
    // 获取文本值
    String text = driver.switchTo().alert().getText();

    // 弹窗确认
    driver.switchTo().alert().accept();
    // 弹窗关闭
    driver.switchTo().alert().dismiss();

    // 输入文本值(仅限Prompt)
    driver.switchTo().alert().sendKeys("xxxxxxxx");
}

上传文件

/* 示例: 原生上传 */
@Test
public void test() {

    /// 示例基于标准上传控件
    // input(type=file)

    driver.get("https://image.baidu.com/");

    WebElement stfile = driver.findElement(By.id("stfile"));
    stfile.sendKeys("./demo.jpg"); // 此方式不会有选择文件的系统弹窗

    driver.close();
}
/* 示例: Robot上传 */
@Test
public void test() {

    /// 示例基于非标准上传控件
    // Robot通过键盘事件操作完成上传文件

    Robot robot = new Robot();
    driver.get("https://image.baidu.com/");

    WebElement sttb = driver.findElement(By.id("sttb"));
    sttb.click();
    WebElement uploadImg = driver.findElement(By.id("uploadImg"));
    uploadImg.click(); // 点击上传按钮, 此时显示选择文件的系统弹窗

    // 复制文件到剪贴板
    StringSelection selection = new StringSelection("./demo.jpg");
    Toolkit.getDefaultToolkit().getSystemClipboard().setContents(selection, null);

    // 点击CTRL+V, 等同于剪贴板的文件地址粘贴到系统弹窗的文件名文本框
    robot.keyPress(KeyEvent.VK_CONTROL);
    robot.keyPress(KeyEvent.VK_V);
    robot.delay(100);
    robot.keyRelease(KeyEvent.VK_CONTROL);
    robot.keyRelease(KeyEvent.VK_V);
    robot.delay(100);

    // 点击ENTER, 等同于点击系统弹窗的打开按钮
    robot.keyPress(KeyEvent.VK_ENTER);
    robot.delay(100);
    robot.keyRelease(KeyEvent.VK_ENTER);

    driver.close();
}

Cookie 操作

/* 示例: Cookie操作 */
@Test
public void test() {

    driver.get("http://news.anhuai.com/");
    
    // 获取所有Cookie
    Set<Cookie> cookies = driver.manage().getCookies();
    // 删除所有Cookie
    driver.manage().deleteAllCookies();

    Cookie cookie = new Cookie("JSESSIONID", //
                               "CD3027F53873A99C1EDA845E5D009B68", //
                               "news.anhuai.com", "/", //
                               DateUtils.addHours(new Date(), 2), //
                               false, true);
    // 新增Cookie
    driver.manage().addCookie(cookie);

    driver.close();
}

截图操作

在自动化测试过程中, 运行失败时通过截图可以很好的定位问题, 因此, 截图操作是自动化测试中的一个重要环节

/* 示例: TakesScreenshot实现截图 */
@Test
public void test() {

    /// 截取浏览器窗体中的内容, 不包括浏览器的菜单和桌面的任务栏区域

    driver.get("http://news.anhuai.com/");

    File screen = ((TakesScreenshot)driver).getScreenshotAs(OutputType.FILE);
    FileCopyUtils.copy(screen, new File("./save/screenshot.jpg"));

    driver.close();
}
/* 示例: Robot实现截图 */
@Test
public void test() {

    /// 截取浏览器窗体中的内容, 以及浏览器的菜单和桌面的任务栏区域

    driver.get("http://news.anhuai.com/");

    BufferedImage image = new Robot().createScreenCapture(
        new Rectangle(Toolkit.getDefaultToolkit().getScreenSize()));
    ImageIO.write(image, "jpg", new File("./save/screenshot.jpg"));

    driver.close();
}
/* 示例: 页面元素截图 */
@Test
public void test() {

    /// 通过指定元素及区域大小截图

    driver.get("http://news.anhuai.com/");

    // 获取要捕捉的元素
    WebElement head = driver.findElement(By.id("head"));

    // 截图整个页面
    File screen = ((TakesScreenshot)driver).getScreenshotAs(OutputType.FILE);
    BufferedImage image = ImageIO.read(screen);

    // 元素的坐标和高宽截取图片
    BufferedImage dest = image.getSubimage(
        head.getLocation().x, head.getLocation().y, 
        head.getSize().width, head.getSize().height);

    ImageIO.write(dest, "jpg",  new File("./save/screenshot.jpg"));

    driver.close();
}

录屏操作

有时未必能通过日志文件或测试报告中操作失败的截图来分析故障, 定位并复现问题, 对于这种情况, 如果可以捕获完整的执行视频, 将有助于定位导致操作失败的问题

示例使用 Monte 媒体库实现视频的录制

<!-- 配置: pom.xml添加monte依赖 -->
<dependency>
    <groupId>com.github.stephenc.monte</groupId>
    <artifactId>monte-screen-recorder</artifactId>
    <version>${monte.version}</version>
</dependency>
/* 示例: 录屏操作 */
public class BrowserTest {

    /// 录制文件默认保存在用户Home目录
    // ~/Viedo/

    private static ScreenRecorder recorder;

    @BeforeAll
    public static void init() {
        // 显卡配置
        GraphicsConfiguration config =
            GraphicsEnvironment
            .getLocalGraphicsEnvironment()
            .getDefaultScreenDevice()
            .getDefaultConfiguration();

        recorder = new ScreenRecorder(config, 
            // 文件格式
            new Format(MediaTypeKey, MediaType.FILE, 
                MimeTypeKey, MIME_AVI), 
            // 视频压缩配置
            new Format(MediaTypeKey, MediaType.VIDEO, 
                EncodingKey, VideoFormatKeys.ENCODING_AVI_TECHSMITH_SCREEN_CAPTURE, 
                CompressorNameKey, VideoFormatKeys.ENCODING_AVI_TECHSMITH_SCREEN_CAPTURE, 
                DepthKey, 24, 
                FrameRateKey, Rational.valueOf(15), 
                QualityKey, 1.0f, 
                KeyFrameIntervalKey, 15 * 60), 
            // 鼠标颜色和刷新速率
            new Format(MediaTypeKey, MediaType.VIDEO, 
                EncodingKey, "black", 
                FrameRateKey, Rational.valueOf(30)), 
            // 音频压缩配置, null不会记录音频
            null);
        
        // 开始录制
        recorder.start();
    }

    @AfterAll
    public static void destroy() {
        // 停止录制
        recorder.stop();
    }

    @Test
    public void test() {

        driver.get("http://news.anhuai.com/");
        Thread.sleep(3000);

        driver.close();
    }
}
